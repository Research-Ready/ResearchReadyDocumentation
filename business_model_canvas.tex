\chapter{Business Model Canvas}\label{business-model-canvas}

\section{Key Partners}\label{key-partners}

\begin{itemize}
\tightlist
\item
  Universities and applied science labs\\
\item
  Research groups focused on reproducibility\\
\item
  Industry partners providing datasets\\
\item
  Cloud or on-prem compute providers\\
\item
  Open-source AI communities\\
\item
  Tooling partners (n8n, Quarto, Jupyter, OpenWebUI)
\end{itemize}

\section{Key Activities}\label{key-activities}

\begin{itemize}
\tightlist
\item
  Building AI-assisted research workflows\\
\item
  Automating reporting and analysis\\
\item
  Integrating RStudio, Jupyter, n8n, OpenWebUI\\
\item
  Maintaining reproducible pipelines\\
\item
  Training researchers on tools and AI adoption\\
\item
  Supporting institutions with deployments
\end{itemize}

\section{Key Resources}\label{key-resources}

\begin{itemize}
\tightlist
\item
  AI stack (OpenWebUI, embeddings, automation workflows)\\
\item
  Expertise in research design and reproducible workflows\\
\item
  On-premise server infrastructure\\
\item
  Quarto templates and automation scripts\\
\item
  Network of collaborators (VCH, Windesheim, NICE, etc.)
\end{itemize}

\section{Value Proposition}\label{value-proposition}

\begin{itemize}
\tightlist
\item
  Faster, cleaner, and repeatable research workflows\\
\item
  Automated reporting that eliminates manual work\\
\item
  Clear reproducibility and version control\\
\item
  AI systems tailored to academic and research contexts\\
\item
  Reduced tool fragmentation for institutions\\
\item
  Structured research environment for students
\end{itemize}

\section{Customer Relationships}\label{customer-relationships}

\begin{itemize}
\tightlist
\item
  High-touch onboarding and consulting\\
\item
  Long-term support contracts\\
\item
  Close collaboration with research teams\\
\item
  Automated systems reducing daily support load\\
\item
  Documentation-first approach\\
\item
  Community channels for ongoing updates
\end{itemize}

\section{Channels}\label{channels}

\begin{itemize}
\tightlist
\item
  Direct outreach to research institutions\\
\item
  Demonstration workshops and training days\\
\item
  LinkedIn and professional networks\\
\item
  GitHub repositories and open-source visibility\\
\item
  Conference presentations\\
\item
  Word of mouth through institutional partners
\end{itemize}

\section{Customer Segments}\label{customer-segments}

\begin{itemize}
\tightlist
\item
  Research groups in applied sciences\\
\item
  Lectoren / professors running research programs\\
\item
  Universities wanting internal AI systems\\
\item
  SMEs needing reproducible sustainability reporting\\
\item
  NGOs and government-linked research units
\end{itemize}

\section{Cost Structure}\label{cost-structure}

\begin{itemize}
\tightlist
\item
  Server and infrastructure maintenance\\

  Development time for automation and AI workflows\\

  Integration and custom feature development\\
\item
  Occasional cloud compute\\

  Administrative overhead
